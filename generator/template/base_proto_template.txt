// BaseResponse 所有 RPC 接口的统一基础响应。
message BaseResponse {
  int32 code = 1; // 状态码
  string message = 2; // 响应消息
}

// OperationResponse 创建、修改、删除等写操作的统一响应。
message OperationResponse {
  BaseResponse baseResp = 1; // 基础响应
}

// PageRequest 通用分页请求。
message PageRequest {
  int64 pageSize = 1; // 每页记录数
  int64 pageNumber = 2; // 当前页码
}

// PageInfo 通用分页信息。
message PageInfo {
  int64 total = 1; // 总记录数
  int64 pageSize = 2; // 每页记录数
  int64 pageNumber = 3; // 当前页码
}
